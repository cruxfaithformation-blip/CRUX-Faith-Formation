rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /videos/{videoId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /spotify/{playlistId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /applemusic/{playlistId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /books/{bookId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /users/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if request.auth != null && request.auth.uid == userId;
    }

    match /userProgress/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if request.auth != null && request.auth.uid == userId;
      allow delete: if request.auth != null && request.auth.uid == userId;

      match /studyReflections/{studyId} {
        allow read: if request.auth != null && request.auth.uid == userId;
        allow create: if request.auth != null && request.auth.uid == userId;
        allow update: if request.auth != null && request.auth.uid == userId;
        allow delete: if request.auth != null && request.auth.uid == userId;
      }

      // ── NEW: Child accounts subcollection ──────────────
      match /children/{childId} {
        allow read: if request.auth != null && request.auth.uid == userId;
        allow create: if request.auth != null && request.auth.uid == userId;
        allow update: if request.auth != null && request.auth.uid == userId;
        allow delete: if request.auth != null && request.auth.uid == userId;

        // ── NEW: Child study journal subcollection ─────────
        match /studyReflections/{studyId} {
          allow read: if request.auth != null && request.auth.uid == userId;
          allow create: if request.auth != null && request.auth.uid == userId;
          allow update: if request.auth != null && request.auth.uid == userId;
          allow delete: if request.auth != null && request.auth.uid == userId;
        }
      }
      // ── END NEW ────────────────────────────────────────
    }

    match /moduleProgress/{docId} {
      allow read: if request.auth != null && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
    }

    match /teachers/{teacherId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == teacherId;
      allow update: if request.auth != null;
    }

    match /profiles/{userId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    match /xpAwards/{awardId} {
      allow read: if request.auth != null && (
        resource.data.studentId == request.auth.uid ||
        resource.data.teacherId == request.auth.uid
      );
      allow write: if request.auth != null;
    }
  }
}
